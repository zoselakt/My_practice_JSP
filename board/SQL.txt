CREATE SEQUENCE BOARD_SEQ
	START WITH 1
	INCREMENT BY 1
	MAXVALUE 99999
	NOCACHE
	NOCYCLE
	NOORDER;

CREATE TABLE BOARD(
	NUM NUMBER PRIMARY KEY,
	TITLE VARCHAR2(50) NOT NULL,
	WRITER VARCHAR2(50) NOT NULL,
	CONTENTS VARCHAR2(1000),
	REGDATE DATE,
	CNT NUMBER DEFAULT 0
);
